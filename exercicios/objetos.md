2. Conta no banco

```javascript
let contaCorrente = {
  saldo: 1000,
  taxaDeJuros: 1,
  depositar: function adicionarDinheiros(qtde) {
    if (qtde > 0) {
      contaCorrente.saldo += qtde;
    }
  },
  sacar: function retirarDinheiro(qtde) {
    var saldoDisponivel = contaCorrente.saldo - qtde;
    if (qtde > 0 && saldoDisponivel >= 0) {
      contaCorrente.saldo -= qtde;
    }
  },
};
```
